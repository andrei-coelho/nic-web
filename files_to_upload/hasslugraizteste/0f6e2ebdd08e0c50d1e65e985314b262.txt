- criar uma imagem
docker build -t my_image . (path/to/files)

- lista todas as imagens
docker images

- remove uma imagem (flag -f para forçar)
docker rmi my_image -f OR docker image rm my_image -f

- criar um container que copia arquivos que são alterados no diretório
docker run --name my_container -p 0000:0000 -v C:/dir/to/changes:workdir -d my_image

- lista todos os containers 
docker ps -a

- remove um container
docker rm my_container OR docker container rm my_container

- executa uma aplicação
docker start my_container

- pausa uma aplicação
docker stop my_container

- usar um shell dentro do container

- executa um comando no bash dentro da aplicação
docker exec -ti my_container sh -c "command"
